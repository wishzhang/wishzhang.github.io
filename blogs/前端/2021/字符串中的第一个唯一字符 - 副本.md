下面给出哈希表解法的 JavaScript 代码实现：

```javascript
/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var firstUniqChar = function (s) {

  let h1 = Array.apply(null, {length: 26}).map(() => 0);

  for (let i = 0; i < s.length; i++) {
    let ch = s[i];
    let index = getHIndex(ch);
    h1[index]++;
  }

  for (let i = 0; i < s.length; i++) {
    let ch = s[i];
    let index = getHIndex(ch);
    if (h1[index] === 1) {
      return i;
    }
  }

  return -1;
};

function getHIndex(ch = '') {
  return ch.charCodeAt() - 97;
}
```
